# Oops! We couldn't find that page.

<script>
    var baseUrl = window.location.origin + '/';
    window.onload = function() {
        let currentUrl = window.location.href;
        if (currentUrl != currentUrl.toLowerCase()) {
            location.replace(currentUrl.toLowerCase());
        }
    };
</script>
